{{- /* Code block render hook — wraps every fenced ```block``` in the .kt-codeblock BEM contract (docs/contracts/components.md), matching shortcodes/code-tabs.html for single-file blocks. Plain Markdown fences would otherwise emit bare Chroma output with no card chrome. Highlighting — including opt-in line numbers ({linenos=table}) and line highlighting ({hl_lines="2 4-5"}) — is delegated to transform.HighlightCodeBlock, which honours .Options merged with the site's [markup.highlight] config (lineNos=false ⇒ numbers are opt-in). A `title=` / `file=` attribute renders the filename as a quiet .kt-codeblock__title label (0.2.0 — this used to be a lone fake tab); terminal languages get the dark --terminal modifier. The copy button is wired by assets/js/theme.js ([data-copy]). */ -}} {{- $lang := .Type | default "text" -}} {{- $terminal := in (slice "bash" "sh" "shell" "zsh" "console" "terminal") (lower $lang) -}} {{- $title := or .Attributes.title .Attributes.file -}}
{{- if $title -}} {{ partial "kt-icon.html" (dict "name" "file-text" "class" "kt-codeblock__file-icon" "size" 13) }}{{ $title }} {{- else if $terminal -}} {{ partial "kt-icon.html" "terminal" }}{{ $lang }} {{- else -}} {{ partial "kt-icon.html" "code" }}{{ $lang }} {{- end -}}
{{- (transform.HighlightCodeBlock .).Wrapped -}}
